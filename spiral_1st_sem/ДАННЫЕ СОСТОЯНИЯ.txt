*   2 - зеркальная версия
*   3 - версия изнутри наружу
*   4 - зеркальная версия изнутри наружу
*/

void printMenu(int statement)
{
    if (statement == 1)
        cout << "ВО ВНУТРЬ ПО ЧАСОВОЙ     <-" << endl;
    else
        cout << "ВО ВНУТРЬ ПО ЧАСОВОЙ" << endl;

    if (statement == 2)
        cout << "ВО ВНУТРЬ ПРОТИВ ЧАСОВОЙ <-" << endl;
    else
        cout << "ВО ВНУТРЬ ПРОТИВ ЧАСОВОЙ" << endl;

    if  (statement == 3)
        cout << "НАРУЖУ ПО ЧАСОВОЙ        <-" << endl;
    else
        cout << "НАРУЖУ ПО ЧАСОВОЙ" << endl;

    if (statement == 4)
        cout << "НАРУЖУ ПРОТИВ ЧАСОВОЙ    <-" << endl;
    else
        cout << "НАРУЖУ ПРОТИВ ЧАСОВОЙ" << endl;
}
